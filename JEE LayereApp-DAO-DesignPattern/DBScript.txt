CREATE TABLE "SYSTEM"."MOVIE_TICKET_BOOKINGS" 
   (	"TICKID" NUMBER(10,0) NOT NULL ENABLE, 
	"CUSTNAME" VARCHAR2(20 BYTE), 
	"TICKETSCOUNT" NUMBER(5,0), 
	"TYPE" VARCHAR2(20 BYTE), 
	"SEATNOS" VARCHAR2(20 BYTE), 
	"BILLAMOUNT" FLOAT(126), 
	 CONSTRAINT "MOVIE_TICKET_BOOKINGS_PK" PRIMARY KEY ("TICKID"))
	 
	 CREATE SEQUENCE  "SYSTEM"."TICKETID_SEQ"  MINVALUE 1000 MAXVALUE 2000 
	                                                               INCREMENT BY 1 START WITH 1000 CACHE 20 NOORDER  NOCYCLE ;
